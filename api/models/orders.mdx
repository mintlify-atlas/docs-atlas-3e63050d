## OT (Orden de Trabajo)

The main interface representing a production work order in the P.FLEX system.

## Usage Example

```typescript
import { OT } from './models/orders.models';

const workOrder: OT = {
  descripcion: 'Etiquetas adhesivas personalizadas',
  'Nro. Cotizacion': 'COT-2026-001',
  'Nro. Ficha': 'FCH-12345',
  Pedido: 'PED-2026-150',
  OT: 'OT-2026-0150',
  'ORDEN COMPRA': 'OC-CLIENT-789',
  'Razon Social': 'Corporación ABC S.A.',
  // ... additional fields
};
```
